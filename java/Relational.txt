class Relational{
	public static void main(string arg[])
	{
		int no1=20;
		int no2=35;

		system.out.println("Is no1>no2: "+(no1>no2));
		system.out.println("Is no1<no2: "+(no1<no2));
		system.out.println("Is no1>=no2: "+(no1>=no2));
		system.out.println("Is no1<=no2: "+(no1<=no2));
		system.out.println("Is no1==no2: "+(no1==no2));
		system.out.println("Is no1!=no2: "+(no1!=no2));

	}
}
